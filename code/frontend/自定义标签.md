```js
(function(){
   var store = new Store();
   var customElement = Object.create(HTMLElement.prototype);
    customElement.createdCallback = function(){
      this.data = store.fetch()
    }
    document.registerElement("my-component",{
      prototype:customElement
    });
  var ele = document.createElement("my-component");
  document.body.appendChild(ele);
})()
```
