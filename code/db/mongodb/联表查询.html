# 创建 Schema

const UserSchema = mongoose.Schem({
  name: String,
  age: Number,
});

const User = mongoose.model('user', UserSchema);

const ArticalSchema = mongoose.Schem({
  content: String,
  title: String,
  // 关联user
  user: {
    type: mongoose.Schem.Types.ObjectId,
    ref: 'user',
  },
});

const Artical = mongoose.model('artical', ArticalSchema);
